      } else {
        altura <- NA
      }
    }
  }
  return(altura)
}

10.3.2 DAP

pegadap <- function(x) {
  x <- gsub("\\+/-", "", x)
  x <- gsub("  ", " ", x)
  x <- gsub("  ", " ", x)
  # print(x)
  pt1 <- "x [-+]?[0-9]*[,.]?[0-9] cm de circ"
  pt2 <- "x [-+]?[0-9]*[,.]?[0-9] cm de di"
  pt3 <- "x [-+]?[0-9]*[,.]?[0-9]cm de di"
  pt4 <- "x [-+]?[0-9]*[,.]?[0-9]cm di"
  pt5 <- "m x [-+]?[0-9]*[,.]?[0-9] cm DAP"
  pt6 <- "m x [-+]?[0-9]*[,.]?[0-9] cm"
  pt7 <- "de [-+]?[0-9]*[,.]?[0-9] m de DAP"
  pt8 <- "[-+]?[0-9]*[,.]?[0-9] cm de DAP"
  pt9 <- "[-+]?[0-9]*[,.]?[0-9] cm (DAP)"
  pt10 <- "[-+]?[0-9]*[,.]?[0-9] cm dbh"
  pt11 <- "DAP [-+]?[0-9]*[,.]?[0-9] cm"
  pt12 <- "[-+]?[0-9]*[,.]?[0-9] cm D.A.P."
  pt13 <- "D.A.P. = [-+]?[0-9]*[,.]?[0-9] cm"
  pt14 <- "[-+]?[0-9]*[,.]?[0-9]cm de di"
  pt15 <- "[-+]?[0-9]*[,.]?[0-9]cm dap"
  pt16 <- "[-+]?[0-9]*[,.]?[0-9]cm. dia"
  pt17 <- "dbh. [-+]?[0-9]*[,.]?[0-9]cm"
  pt18 <- "[-+]?[0-9]*[,.]?[0-9]cm. in dia"
  pt19 <- "[-+]?[0-9]*[,.]?[0-9] cm de di"
  pt20 <- "[-+]?[0-9]*[,.]?[0-9] cm (DAP)"
  pt21 <- "DBH [-+]?[0-9]*[,.]?[0-9] cm"

  # x="Árvore de 13m de altura x 11cm de diâmetro do fuste."
  altura <- NA
  for (p in 1:21) {
    pt <- get(paste("pt", p, sep = ""))
    gp <- grep(pt, x, ignore.case = F)
    if (length(gp) == 1 & is.na(altura)) {
      # print(p)
      rmm <- strsplit(x, pt)[[1]]
      rmm <- rmm[rmm != "" & rmm != "." & !is.na(rmm)]
      xx <- x
      if (length(rmm) == 1) {
        for (r in length(rmm):1) {
          xx <- gsub(rmm[r], "", xx, fixed = T, useBytes = T)
        }
      } else {
        if (length(rmm) == 2) {
          n1 <- nchar(rmm[1])
          n2 <- nchar(rmm[2])
          n0 <- nchar(x)
          ns <- n1 + 1
          nt <- n0 - n2
          xx <- substr(x, ns, nt)
        } else {
          if (length(rmm) > 2) {
            xx <- NA
          }
        }
      }
      if (!is.na(xx)) {
        xx <- trimws(gsub("[A-Z]|\\(|\\)|:|=", "", xx, ignore.case = T), which = "both")
        xx <- gsub(",", ".", xx)
        tt <- grep("-", xx)
        if (length(tt) > 0) {
          xxx <- strsplit(xx, "-")[[1]]
          xxx <- xxx[xxx != ""]
          xxx <- gsub(" \\.", "", xxx)
          xx <- mean(as.numeric(trimws(xxx, which = "both")), na.rm = T)
        } else {
          xx <- strsplit(xx, " ")[[1]]
          xx <- trimws(xx, which = "both")
          xx <- xx[xx != "." & xx != "" & xx != "..."]
          xx <- xx[1]
          if (substr(xx, nchar(xx), nchar(xx)) == ".") {
            xx <- substr(xx, 1, nchar(xx) - 1)
          }
        }
        xx <- as.numeric(xx)
        if (xx > 0) {
          altura <- xx
        }
      }
    }
  }
  return(altura)
}
